//Write a program that checks if the string contains all digits using Regular Expression.

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Text.RegularExpressions;

namespace MoreCodesCSharpConsole
{
    class Program
    {
        static void Main(string[] args)
        {
            Regex regex = new Regex("\\d+");
            string value = "12345";

            Console.Write(regex.IsMatch(value));
            Console.ReadKey();
        }
    }
}
